## 二、代码重复

### 2.1 重复的错误处理逻辑
**位置**: `DetailView.vue`

**现状**:
```typescript
// 每个操作都有类似的错误处理
const toggleFavorite = async () => {
  try {
    await libraryService.toggleFavorite(...)
  } catch (err) {
    if (err instanceof HttpClientError) {
      patchError.value = err.apiError?.message || err.message
    } else if (err instanceof Error) {
      patchError.value = err.message
    }
  }
}

// 同样逻辑在 updateUserRating, updateUserTags 等中重复
```

**改进建议**:
```typescript
// 提取通用错误处理器
function useErrorHandler() {
  const error = ref('')

  const handleError = (err: unknown, fallbackKey: string) => {
    const message = err instanceof HttpClientError
      ? err.apiError?.message || err.message
      : err instanceof Error
        ? err.message
        : t(fallbackKey)
    error.value = message
    return message
  }

  const clearError = () => { error.value = '' }

  return { error, handleError, clearError }
}

// 使用
const { error: patchError, handleError } = useErrorHandler()

const toggleFavorite = async () => {
  try {
    await libraryService.toggleFavorite(...)
  } catch (err) {
    handleError(err, 'detail.errFavorite')
  }
}
```

**严重程度**: 🟡 中等

### 2.2 重复的 watchDebounced 模式
**位置**: `AppShell.vue`

**现状**:
```typescript
// 三个几乎相同的 watchDebounced 块
watchDebounced(searchDraft, handler1, { debounce: 280 })
watchDebounced(searchDraftFrames, handler2, { debounce: 280 })
watchDebounced(searchDraftActors, handler3, { debounce: 280 })
```

**改进建议**:
```typescript
// 提取通用搜索 composable
function useSearchQuery(
  routeQueryKey: string,
  routeName: string,
  options: { debounce: number }
) {
  const draft = ref('')

  watchDebounced(
    draft,
    (value) => {
      const normalized = value.trim()
      const current = (route.query[routeQueryKey] as string)?.trim() || ''
      if (normalized === current) return

      router.replace({
        name: routeName,
        query: { ...route.query, [routeQueryKey]: normalized || undefined }
      })
    },
    { debounce: options.debounce }
  )

  return { draft }
}

// 使用
const { draft: searchDraft } = useSearchQuery('q', 'library', { debounce: 280 })
```

**严重程度**: 🟢 轻微
