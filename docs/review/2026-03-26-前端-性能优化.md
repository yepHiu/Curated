### 1.2 图片懒加载与优化（已完成）

#### 问题：海报图片加载策略
**位置**: `MovieCard.vue`, `DetailPage.vue`

**现状问题**:
```typescript
// 直接加载原图，无尺寸控制
<img :src="movie.coverUrl" />

// 没有懒加载，首屏加载所有图片
// 没有占位符，布局跳动 (CLS)
```

**优化方案**:
```vue
<template>
  <div class="aspect-[2/3] relative">
    <!-- 骨架屏占位 -->
    <Skeleton v-if="!loaded" class="absolute inset-0" />

    <img
      ref="imgRef"
      :src="optimizedSrc"
      :srcset="srcSet"
      :alt="movie.title"
      loading="lazy"
      decoding="async"
      class="object-cover transition-opacity duration-300"
      :class="loaded ? 'opacity-100' : 'opacity-0'"
      @load="loaded = true"
    />
  </div>
</template>

<script setup>
import { useImage } from '@vueuse/core'

const props = defineProps<{ movie: Movie }>()

// 使用 @vueuse/core 的 useImage
const { isLoading, error } = useImage({
  src: computed(() => props.movie.coverUrl)
})

// 生成响应式 srcset
const srcSet = computed(() => {
  const base = props.movie.coverUrl
  return `
    ${base}?w=200 200w,
    ${base}?w=400 400w,
    ${base}?w=800 800w
  `
})

// 使用较小的缩略图作为占位
const optimizedSrc = computed(() => {
  return `${props.movie.thumbUrl || props.movie.coverUrl}?w=400`
})
</script>
```

**额外优化**:
```typescript
// 使用 Intersection Observer 精确控制懒加载
import { useIntersectionObserver } from '@vueuse/core'

const target = ref(null)
const shouldLoad = ref(false)

useIntersectionObserver(
  target,
  ([{ isIntersecting }]) => {
    if (isIntersecting) {
      shouldLoad.value = true
    }
  },
  { rootMargin: '100px' }  // 提前 100px 开始加载
)
```

**预期收益**: 首屏图片请求减少 70%，LCP 降低 1-2s

**优先级**: 🔴 高

**优化方案**:
```typescript
// 1. 请求去重
const pendingRequests = new Map<string, Promise<any>>()

export function dedupedRequest<T>(key: string, request: () => Promise<T>): Promise<T> {
  if (pendingRequests.has(key)) {
    return pendingRequests.get(key)!
  }

  const promise = request().finally(() => {
    pendingRequests.delete(key)
  })

  pendingRequests.set(key, promise)
  return promise
}

// 2. 增量更新（使用 if-modified-since 或 etag）
async function fetchMoviesIncremental(lastSync: string) {
  const response = await api.listMovies({
    modifiedSince: lastSync,
    fields: ['id', 'updatedAt']  // 只取变更的 ID
  })

  // 只请求变更的详情
  const changedIds = response.items.filter(item => isChanged(item))
  const details = await Promise.all(
    changedIds.map(id => api.getMovieDetail(id))
  )

  return details
}

// 3. 请求合并（debounce）
import { debounce } from 'lodash-es'

const debouncedReload = debounce(reloadMoviesFromApi, 300, {
  leading: false,
  trailing: true
})
```

**预期收益**: 减少 50% 无效请求

### 2.2 状态更新优化

#### 问题：大数据量更新卡顿
**位置**: `web-library-service.ts:124-147`

**现状**:
```typescript
function mergeMovieIntoListState(movie: Movie) {
  // 每次更新都创建新数组，大数组时性能差
  moviesState.value = moviesState.value.map((m, i) =>
    i === idx ? { ...m, ...movie } : m
  )
}
```

**优化方案**:
```typescript
import { shallowRef, triggerRef } from 'vue'

// 使用 shallowRef + 手动触发
const moviesState = shallowRef<Movie[]>([])

function mergeMovieIntoListState(movie: Movie) {
  const idx = moviesState.value.findIndex(m => m.id === movie.id)
  if (idx >= 0) {
    // 直接修改，不创建新数组
    Object.assign(moviesState.value[idx], movie)
    // 只触发一次更新
    triggerRef(moviesState)
  }
}

// 或使用 Map 存储
const moviesMap = ref<Map<string, Movie>>(new Map())

function mergeMovie(movie: Movie) {
  moviesMap.value.set(movie.id, movie)
}

// 视图使用 computed
const moviesList = computed(() => Array.from(moviesMap.value.values()))
```

**预期收益**: 大列表更新 FPS 提升 3-5 倍

**优先级**: 🟡 中

### 2.3 预加载策略

#### 实现智能预加载
```typescript
// 预加载下一页数据
const usePrefetch = () => {
  const prefetchNextPage = async (currentOffset: number) => {
    const nextOffset = currentOffset + PAGE_SIZE
    const cacheKey = `movies-${nextOffset}`

    if (!queryClient.getQueryData(cacheKey)) {
      await queryClient.prefetchQuery({
        queryKey: [cacheKey],
        queryFn: () => api.listMovies({ offset: nextOffset }),
        staleTime: 5 * 60 * 1000
      })
    }
  }

  // 预加载详情（hover 时）
  const prefetchDetail = (movieId: string) => {
    queryClient.prefetchQuery({
      queryKey: ['movie', movieId],
      queryFn: () => api.getMovieDetail(movieId),
      staleTime: 10 * 60 * 1000
    })
  }

  return { prefetchNextPage, prefetchDetail }
}
```

**预期收益**: 页面切换时间减少 60%

**优先级**: 🟢 低
